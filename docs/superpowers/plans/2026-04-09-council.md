# Council Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a persistent advisory boardroom where the main agent can convene specialist agents for collaborative planning discussions, with live streaming in the web UI.

**Architecture:** New `pkg/council/` package with its own orchestration loop that calls LLM providers directly. A `council` tool lets the main agent trigger councils. Council events flow through the EventBus to the Pico channel for live web UI rendering. Sessions persist as JSONL transcripts with JSON metadata.

**Tech Stack:** Go (backend engine, tool, API, Pico streaming), React 19 + TanStack Router + Jotai (frontend), WebSocket (live streaming)

package council

import "time"

// Council session statuses.
const (
	StatusActive = "active"
	StatusPaused = "paused"
	StatusClosed = "closed"
)

// Transcript entry roles.
const (
	RoleSystem    = "system"
	RoleAgent     = "agent"
	RoleModerator = "moderator"
	RoleUser      = "user"
	RoleSynthesis = "synthesis"
)

// CouncilMeta holds the metadata for a council session.
type CouncilMeta struct {
	ID          string    `json:"id"`
	Title       string    `json:"title"`
	Description string    `json:"description,omitempty"`
	Roster      []string  `json:"roster"`
	Status      string    `json:"status"`
	Rounds      int       `json:"rounds"`
	CreatedAt   time.Time `json:"created_at"`
	UpdatedAt   time.Time `json:"updated_at"`
}

// TranscriptEntry is a single message in the council transcript.
type TranscriptEntry struct {
	Role      string    `json:"role"`
	AgentID   string    `json:"agent_id,omitempty"`
	AgentType string    `json:"agent_type,omitempty"`
	Content   string    `json:"content"`
	Round     int       `json:"round"`
	Timestamp time.Time `json:"ts"`
}

// CouncilConfig is the input to start or resume a council.
type CouncilConfig struct {
	ID          string   // set for resume, empty for new
	Title       string
	Description string
	Topic       string   // opening prompt or new agenda item
	Roster      []string // agent type names
	Model       string   // moderator model
}

// CouncilResult is returned when a council concludes.
type CouncilResult struct {
	ID        string
	Title     string
	Rounds    int
	Synthesis string
	Status    string
}

package council

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"sort"
	"sync"
	"time"

	"github.com/dawnforge-lab/spawnbot-v5/pkg/logger"
)

// Store handles council session persistence.
type Store struct {
	dir string
	mu  sync.RWMutex
}

// NewStore creates a council store rooted at dir.
func NewStore(dir string) *Store {
	return &Store{dir: dir}
}

// Create initializes a new council session on disk and returns its ID.
func (s *Store) Create(meta *CouncilMeta) (string, error) {
	s.mu.Lock()
	defer s.mu.Unlock()

	now := time.Now()
	meta.ID = fmt.Sprintf("council-%d", now.UnixMilli())
	meta.CreatedAt = now
	meta.UpdatedAt = now

	councilDir := filepath.Join(s.dir, meta.ID)
	if err := os.MkdirAll(councilDir, 0o755); err != nil {
		return "", fmt.Errorf("create council dir: %w", err)
	}

	if err := s.writeMeta(meta); err != nil {
		os.RemoveAll(councilDir)
		return "", err
	}

	return meta.ID, nil
}

// Load reads council metadata from disk.
func (s *Store) Load(id string) (*CouncilMeta, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()

	metaPath := filepath.Join(s.dir, id, "meta.json")
	data, err := os.ReadFile(metaPath)
	if err != nil {
		return nil, fmt.Errorf("read council meta: %w", err)
	}

	var meta CouncilMeta
	if err := json.Unmarshal(data, &meta); err != nil {
		return nil, fmt.Errorf("parse council meta: %w", err)
	}
	return &meta, nil
}

// SaveMeta writes updated metadata to disk.
func (s *Store) SaveMeta(meta *CouncilMeta) error {
	s.mu.Lock()
	defer s.mu.Unlock()

	meta.UpdatedAt = time.Now()
	return s.writeMeta(meta)
}

// List returns all council metadata sorted by updated_at descending.
func (s *Store) List() ([]*CouncilMeta, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()

	entries, err := os.ReadDir(s.dir)
	if err != nil {
		if os.IsNotExist(err) {
			return nil, nil
		}
		return nil, fmt.Errorf("list councils: %w", err)
	}

	var list []*CouncilMeta
	for _, entry := range entries {
		if !entry.IsDir() {
			continue
		}
		metaPath := filepath.Join(s.dir, entry.Name(), "meta.json")
		data, err := os.ReadFile(metaPath)
		if err != nil {
			continue
		}
		var meta CouncilMeta
		if err := json.Unmarshal(data, &meta); err != nil {
			continue
		}
		list = append(list, &meta)
	}

	sort.Slice(list, func(i, j int) bool {
		return list[i].UpdatedAt.After(list[j].UpdatedAt)
	})
	return list, nil
}

// Delete removes a council session from disk.
func (s *Store) Delete(id string) error {
	s.mu.Lock()
	defer s.mu.Unlock()

	councilDir := filepath.Join(s.dir, id)
	return os.RemoveAll(councilDir)
}

// AppendMessage appends a transcript entry to the council's JSONL file.
func (s *Store) AppendMessage(id string, entry TranscriptEntry) {
	if entry.Timestamp.IsZero() {
		entry.Timestamp = time.Now()
	}

	data, err := json.Marshal(entry)
	if err != nil {
		logger.ErrorCF("council", "marshal transcript entry", map[string]any{"error": err.Error()})
		return
	}

	transcriptPath := filepath.Join(s.dir, id, "transcript.jsonl")
	f, err := os.OpenFile(transcriptPath, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0o644)
	if err != nil {
		logger.ErrorCF("council", "open transcript", map[string]any{"error": err.Error()})
		return
	}
	defer f.Close()

	f.Write(append(data, '\n'))
}

// GetTranscript reads all transcript entries for a council.
func (s *Store) GetTranscript(id string) ([]TranscriptEntry, error) {
	s.mu.RLock()
	defer s.mu.RUnlock()

	transcriptPath := filepath.Join(s.dir, id, "transcript.jsonl")
	data, err := os.ReadFile(transcriptPath)
	if err != nil {
		if os.IsNotExist(err) {
			return nil, nil
		}
		return nil, fmt.Errorf("read transcript: %w", err)
	}

	var entries []TranscriptEntry
	for _, line := range splitJSONL(data) {
		var entry TranscriptEntry
		if err := json.Unmarshal(line, &entry); err != nil {
			continue
		}
		entries = append(entries, entry)
	}
	return entries, nil
}

func (s *Store) writeMeta(meta *CouncilMeta) error {
	data, err := json.MarshalIndent(meta, "", "  ")
	if err != nil {
		return fmt.Errorf("marshal council meta: %w", err)
	}
	metaPath := filepath.Join(s.dir, meta.ID, "meta.json")
	return os.WriteFile(metaPath, data, 0o644)
}

// splitJSONL splits byte data into non-empty lines.
func splitJSONL(data []byte) [][]byte {
	var lines [][]byte
	start := 0
	for i := 0; i < len(data); i++ {
		if data[i] == '\n' {
			line := data[start:i]
			if len(line) > 0 {
				lines = append(lines, line)
			}
			start = i + 1
		}
	}
	if start < len(data) {
		line := data[start:]
		if len(line) > 0 {
			lines = append(lines, line)
		}
	}
	return lines
}

package council

import (
	"context"
	"fmt"
	"strings"

	"github.com/dawnforge-lab/spawnbot-v5/pkg/agent"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/agents"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/logger"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/providers"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/providers/protocoltypes"
)

// Engine orchestrates multi-agent council discussions.
type Engine struct {
	store         *Store
	agentRegistry *agents.Registry
	provider      providers.LLMProvider
	eventBus      *agent.EventBus
}

// NewEngine creates a council engine. eventBus may be nil (events will be skipped).
func NewEngine(
	store *Store,
	agentRegistry *agents.Registry,
	provider providers.LLMProvider,
	eventBus *agent.EventBus,
) *Engine {
	return &Engine{
		store:         store,
		agentRegistry: agentRegistry,
		provider:      provider,
		eventBus:      eventBus,
	}
}

// Run executes a council session (new or resumed).
func (e *Engine) Run(ctx context.Context, cfg CouncilConfig) (*CouncilResult, error) {
	meta, transcript, err := e.initSession(cfg)
	if err != nil {
		return nil, err
	}

	e.emitEvent(agent.EventKindCouncilStart, agent.CouncilStartPayload{
		CouncilID:   meta.ID,
		Title:       meta.Title,
		Description: meta.Description,
		Roster:      meta.Roster,
	})

	// If resuming with a new topic, append it
	if cfg.Topic != "" && cfg.ID != "" {
		entry := TranscriptEntry{
			Role:    RoleUser,
			Content: cfg.Topic,
			Round:   meta.Rounds + 1,
		}
		e.store.AppendMessage(meta.ID, entry)
		transcript = append(transcript, entry)
	}

	// Round loop
	for {
		if err := ctx.Err(); err != nil {
			meta.Status = StatusPaused
			e.store.SaveMeta(meta)
			return e.buildResult(meta, ""), nil
		}

		meta.Rounds++
		round := meta.Rounds

		e.emitEvent(agent.EventKindCouncilRoundStart, agent.CouncilRoundStartPayload{
			CouncilID: meta.ID,
			Round:     round,
		})

		// Each agent speaks
		for _, agentType := range meta.Roster {
			agentDef := e.agentRegistry.Get(agentType)
			if agentDef == nil {
				continue
			}

			e.emitEvent(agent.EventKindCouncilAgentStart, agent.CouncilAgentStartPayload{
				CouncilID: meta.ID,
				AgentID:   agentType,
				AgentType: agentType,
				Round:     round,
			})

			content, err := e.callAgent(ctx, agentDef, meta, transcript, cfg.Model)
			if err != nil {
				logger.ErrorCF("council", "agent call failed", map[string]any{
					"agent": agentType,
					"error": err.Error(),
				})
				content = fmt.Sprintf("[Error: %v]", err)
			}

			entry := TranscriptEntry{
				Role:      RoleAgent,
				AgentID:   agentType,
				AgentType: agentType,
				Content:   content,
				Round:     round,
			}
			e.store.AppendMessage(meta.ID, entry)
			transcript = append(transcript, entry)

			e.emitEvent(agent.EventKindCouncilAgentEnd, agent.CouncilAgentEndPayload{
				CouncilID: meta.ID,
				AgentID:   agentType,
				Content:   content,
				Round:     round,
			})
		}

		// Moderator decides: continue or conclude?
		decision, err := e.moderatorDecision(ctx, meta, transcript, cfg.Model)
		if err != nil {
			decision = "conclude"
		}

		e.emitEvent(agent.EventKindCouncilRoundEnd, agent.CouncilRoundEndPayload{
			CouncilID: meta.ID,
			Round:     round,
			Decision:  decision,
		})

		e.store.SaveMeta(meta)

		if decision == "conclude" {
			break
		}

		// Append moderator's continue note to transcript
		moderatorEntry := TranscriptEntry{
			Role:    RoleModerator,
			Content: decision,
			Round:   round,
		}
		e.store.AppendMessage(meta.ID, moderatorEntry)
		transcript = append(transcript, moderatorEntry)
	}

	// Generate synthesis
	synthesis, err := e.generateSynthesis(ctx, meta, transcript, cfg.Model)
	if err != nil {
		synthesis = "Failed to generate synthesis."
	}

	e.store.AppendMessage(meta.ID, TranscriptEntry{
		Role:    RoleSynthesis,
		Content: synthesis,
		Round:   meta.Rounds,
	})

	meta.Status = StatusClosed
	e.store.SaveMeta(meta)

	e.emitEvent(agent.EventKindCouncilEnd, agent.CouncilEndPayload{
		CouncilID: meta.ID,
		Rounds:    meta.Rounds,
		Synthesis: synthesis,
		Status:    StatusClosed,
	})

	return e.buildResult(meta, synthesis), nil
}

func (e *Engine) initSession(cfg CouncilConfig) (*CouncilMeta, []TranscriptEntry, error) {
	if cfg.ID != "" {
		// Resume existing council
		meta, err := e.store.Load(cfg.ID)
		if err != nil {
			return nil, nil, fmt.Errorf("load council %q: %w", cfg.ID, err)
		}
		meta.Status = StatusActive
		e.store.SaveMeta(meta)

		transcript, err := e.store.GetTranscript(cfg.ID)
		if err != nil {
			return nil, nil, fmt.Errorf("load transcript: %w", err)
		}
		return meta, transcript, nil
	}

	// New council
	meta := &CouncilMeta{
		Title:       cfg.Title,
		Description: cfg.Description,
		Roster:      cfg.Roster,
		Status:      StatusActive,
	}
	if _, err := e.store.Create(meta); err != nil {
		return nil, nil, fmt.Errorf("create council: %w", err)
	}

	var transcript []TranscriptEntry
	if cfg.Topic != "" {
		entry := TranscriptEntry{
			Role:    RoleUser,
			Content: cfg.Topic,
			Round:   0,
		}
		e.store.AppendMessage(meta.ID, entry)
		transcript = append(transcript, entry)
	}

	return meta, transcript, nil
}

func (e *Engine) callAgent(
	ctx context.Context,
	agentDef *agents.AgentDefinition,
	meta *CouncilMeta,
	transcript []TranscriptEntry,
	defaultModel string,
) (string, error) {
	model := agentDef.Model
	if model == "" {
		model = defaultModel
	}

	systemPrompt := fmt.Sprintf(
		"You are %s in a council discussion titled %q.\n%s\n\n"+
			"You are participating in a multi-agent planning discussion. "+
			"Read the transcript and contribute your perspective. Be concise and substantive. "+
			"Build on what others have said. If you have nothing new to add, say so briefly.",
		agentDef.Name, meta.Title, agentDef.SystemPrompt,
	)

	messages := []protocoltypes.Message{
		{Role: "system", Content: systemPrompt},
	}
	for _, entry := range transcript {
		role := "user"
		prefix := ""
		switch entry.Role {
		case RoleAgent:
			role = "assistant"
			if entry.AgentID != agentDef.Name {
				role = "user"
				prefix = fmt.Sprintf("[%s]: ", entry.AgentID)
			}
		case RoleModerator:
			prefix = "[Moderator]: "
		case RoleUser:
			prefix = "[User]: "
		case RoleSynthesis:
			prefix = "[Previous Synthesis]: "
		}
		messages = append(messages, protocoltypes.Message{
			Role:    role,
			Content: prefix + entry.Content,
		})
	}

	// Stream if provider supports it
	var accumulated string
	if sp, ok := e.provider.(providers.StreamingProvider); ok {
		resp, err := sp.ChatStream(ctx, messages, nil, model, nil, func(content string) {
			delta := content[len(accumulated):]
			accumulated = content
			if delta != "" {
				e.emitEvent(agent.EventKindCouncilAgentDelta, agent.CouncilAgentDeltaPayload{
					CouncilID: meta.ID,
					AgentID:   agentDef.Name,
					Delta:     delta,
				})
			}
		})
		if err != nil {
			return "", err
		}
		return resp.Content, nil
	}

	resp, err := e.provider.Chat(ctx, messages, nil, model, nil)
	if err != nil {
		return "", err
	}
	return resp.Content, nil
}

func (e *Engine) moderatorDecision(
	ctx context.Context,
	meta *CouncilMeta,
	transcript []TranscriptEntry,
	model string,
) (string, error) {
	var sb strings.Builder
	for _, entry := range transcript {
		switch entry.Role {
		case RoleAgent:
			fmt.Fprintf(&sb, "[%s]: %s\n\n", entry.AgentID, entry.Content)
		case RoleUser:
			fmt.Fprintf(&sb, "[User]: %s\n\n", entry.Content)
		case RoleModerator:
			fmt.Fprintf(&sb, "[Moderator]: %s\n\n", entry.Content)
		}
	}

	messages := []protocoltypes.Message{
		{
			Role: "system",
			Content: "You are the moderator of a council discussion. " +
				"Review the transcript and decide: should the discussion continue for another round, or has enough ground been covered to conclude?\n\n" +
				"If the discussion should CONTINUE, respond with a brief note on what still needs to be explored.\n" +
				"If the discussion should CONCLUDE, respond with exactly the word CONCLUDE and nothing else.",
		},
		{
			Role:    "user",
			Content: fmt.Sprintf("Council: %s\n\nTranscript:\n%s\n\nShould we continue or conclude?", meta.Title, sb.String()),
		},
	}

	resp, err := e.provider.Chat(ctx, messages, nil, model, nil)
	if err != nil {
		return "conclude", err
	}

	content := strings.TrimSpace(resp.Content)
	if strings.HasPrefix(strings.ToUpper(content), "CONCLUDE") {
		return "conclude", nil
	}
	return content, nil
}

func (e *Engine) generateSynthesis(
	ctx context.Context,
	meta *CouncilMeta,
	transcript []TranscriptEntry,
	model string,
) (string, error) {
	var sb strings.Builder
	for _, entry := range transcript {
		switch entry.Role {
		case RoleAgent:
			fmt.Fprintf(&sb, "[%s]: %s\n\n", entry.AgentID, entry.Content)
		case RoleUser:
			fmt.Fprintf(&sb, "[User]: %s\n\n", entry.Content)
		}
	}

	messages := []protocoltypes.Message{
		{
			Role: "system",
			Content: "You are synthesizing the results of a council discussion. " +
				"Produce a clear, actionable summary of the key decisions, plans, and action items. " +
				"Be concise but thorough.",
		},
		{
			Role:    "user",
			Content: fmt.Sprintf("Council: %s\nDescription: %s\n\nTranscript:\n%s\n\nPlease synthesize the discussion.", meta.Title, meta.Description, sb.String()),
		},
	}

	resp, err := e.provider.Chat(ctx, messages, nil, model, nil)
	if err != nil {
		return "", err
	}
	return resp.Content, nil
}

type CouncilTool struct {

func (t *CouncilTool) Execute(ctx context.Context, args map[string]any) *ToolResult {
	action, _ := args["action"].(string)

	switch action {
	case "list":
		return t.executeList()
	case "start":
		return t.executeStart(ctx, args)
	case "resume":
		return t.executeResume(ctx, args)
	default:
		return ErrorResult(fmt.Sprintf("unknown action %q — use start, resume, or list", action))
	}
}

func (t *CouncilTool) executeList() *ToolResult {
	store := t.getStore()
	if store == nil {
		return ErrorResult("council store not configured")
	}

	list, err := store.List()
	if err != nil {
		return ErrorResult(fmt.Sprintf("list councils: %v", err))
	}

	if len(list) == 0 {
		return &ToolResult{
			ForUser: "No council sessions found.",
			ForLLM:  "No council sessions exist yet. Use action 'start' to create one.",
		}
	}

	var sb strings.Builder
	for _, meta := range list {
		fmt.Fprintf(&sb, "- **%s** (id: %s) — %s, %d rounds, status: %s\n",
			meta.Title, meta.ID, meta.Description, meta.Rounds, meta.Status)
	}

	data, _ := json.Marshal(list)
	return &ToolResult{
		ForUser: fmt.Sprintf("Found %d council session(s):\n%s", len(list), sb.String()),
		ForLLM:  string(data),
	}
}

func (t *CouncilTool) executeStart(ctx context.Context, args map[string]any) *ToolResult {
	title, _ := args["title"].(string)
	if strings.TrimSpace(title) == "" {
		return ErrorResult("title is required for action 'start'")
	}
	description, _ := args["description"].(string)
	topic, _ := args["topic"].(string)

	var roster []string
	if raw, ok := args["roster"].([]any); ok {
		for _, r := range raw {
			if s, ok := r.(string); ok {
				roster = append(roster, s)
			}
		}
	}

	if t.engine == nil {
		return ErrorResult("council engine not configured")
	}

	result, err := t.engine.Run(ctx, council.CouncilConfig{
		Title:       title,
		Description: description,
		Topic:       topic,
		Roster:      roster,
	})
	if err != nil {
		return ErrorResult(fmt.Sprintf("council failed: %v", err))
	}

	return t.formatResult(result)
}

func (t *CouncilTool) executeResume(ctx context.Context, args map[string]any) *ToolResult {
	councilID, _ := args["council_id"].(string)
	if strings.TrimSpace(councilID) == "" {
		return ErrorResult("council_id is required for action 'resume'")
	}
	topic, _ := args["topic"].(string)

	if t.engine == nil {
		return ErrorResult("council engine not configured")
	}

	result, err := t.engine.Run(ctx, council.CouncilConfig{
		ID:    councilID,
		Topic: topic,
	})
	if err != nil {
		return ErrorResult(fmt.Sprintf("council resume failed: %v", err))
	}

	return t.formatResult(result)
}

func (t *CouncilTool) formatResult(result *council.CouncilResult) *ToolResult {
	userMsg := fmt.Sprintf("Council %q completed (%d rounds).\n\n**Synthesis:**\n%s",
		result.Title, result.Rounds, result.Synthesis)

	data, _ := json.Marshal(result)
	return &ToolResult{
		ForUser: userMsg,
		ForLLM:  string(data),
	}
}

func (t *CouncilTool) getStore() *council.Store {
	if t.store != nil {
		return t.store
	}
	if t.storeDir != "" {
		t.store = council.NewStore(t.storeDir)
		return t.store
	}
	return nil
}

func (h *Handler) registerCouncilRoutes(mux *http.ServeMux) {
	mux.HandleFunc("GET /api/councils", h.handleListCouncils)
	mux.HandleFunc("GET /api/councils/{id}", h.handleGetCouncil)
	mux.HandleFunc("DELETE /api/councils/{id}", h.handleDeleteCouncil)

func (h *Handler) handleListCouncils(w http.ResponseWriter, r *http.Request) {
	store := h.getCouncilStore()
	if store == nil {
		http.Error(w, "council store not configured", http.StatusInternalServerError)
		return
	}

	list, err := store.List()
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(list)
}

func (h *Handler) handleGetCouncil(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	store := h.getCouncilStore()
	if store == nil {
		http.Error(w, "council store not configured", http.StatusInternalServerError)
		return
	}

	meta, err := store.Load(id)
	if err != nil {
		http.Error(w, err.Error(), http.StatusNotFound)
		return
	}

	transcript, err := store.GetTranscript(id)
	if err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}

	resp := struct {
		*council.CouncilMeta
		Transcript []council.TranscriptEntry `json:"transcript"`
	}{
		CouncilMeta: meta,
		Transcript:  transcript,
	}

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(resp)
}

func (h *Handler) handleDeleteCouncil(w http.ResponseWriter, r *http.Request) {
	id := r.PathValue("id")
	store := h.getCouncilStore()
	if store == nil {
		http.Error(w, "council store not configured", http.StatusInternalServerError)
		return
	}

	if err := store.Delete(id); err != nil {
		http.Error(w, err.Error(), http.StatusInternalServerError)
		return
	}

	w.WriteHeader(http.StatusNoContent)
}

func (h *Handler) getCouncilStore() *council.Store {
	if h.councilStore != nil {
		return h.councilStore
	}
	// Lazy init from config workspace path
	if h.cfg != nil {
		h.councilStore = council.NewStore(h.cfg.WorkspacePath() + "/councils")
	}
	return h.councilStore
}

					"council_id": p.CouncilID,
					"agent_id":   p.AgentID,
					"content":    p.Content,
					"round":      p.Round,
				}
			case agent.EventKindCouncilRoundEnd:
				p := evt.Payload.(agent.CouncilRoundEndPayload)
				msgType = TypeCouncilRoundEnd
				payload = map[string]any{
					"council_id": p.CouncilID,
					"round":      p.Round,
					"decision":   p.Decision,
				}
			case agent.EventKindCouncilEnd:
				p := evt.Payload.(agent.CouncilEndPayload)
				msgType = TypeCouncilEnd
				payload = map[string]any{
					"council_id": p.CouncilID,
					"rounds":     p.Rounds,
					"synthesis":  p.Synthesis,
					"status":     p.Status,
				}
			default:
				continue
			}

			msg := newMessage(msgType, payload)
			c.broadcastToAll(msg)
		}
	}()
}

// broadcastToAll sends a message to ALL active Pico connections (all sessions).
func (c *PicoChannel) broadcastToAll(msg PicoMessage) {
	c.connsMu.RLock()
	defer c.connsMu.RUnlock()

	for _, bySession := range c.sessionConnections {
		for _, pc := range bySession {
			pc.writeJSON(msg)
		}
	}
}

// BeginStream implements channels.StreamingCapable.
func (c *PicoChannel) BeginStream(ctx context.Context, chatID string) (channels.Streamer, error) {
	if !c.IsRunning() {
		return nil, channels.ErrNotRunning
	}

	return &picoStreamer{
		channel:          c,
		chatID:           chatID,
		messageID:        fmt.Sprintf("stream-%d", time.Now().UnixMilli()),
		throttleInterval: 50 * time.Millisecond,
	}, nil
}

// picoStreamer streams partial LLM output via WebSocket message.update events.
type picoStreamer struct {
	channel          *PicoChannel
	chatID           string
	messageID        string
	throttleInterval time.Duration
	lastAt           time.Time
	mu               sync.Mutex
	sent             bool
}

func (s *picoStreamer) Update(ctx context.Context, content string) error {
	s.mu.Lock()
	defer s.mu.Unlock()

	now := time.Now()
	if s.sent && now.Sub(s.lastAt) < s.throttleInterval {
		return nil
	}

	msg := newMessage(TypeMessageUpdate, map[string]any{
		"message_id": s.messageID,
		"content":    content,
	})

	// If first update, send a message.create first so frontend has the message
	if !s.sent {
		createMsg := newMessage(TypeMessageCreate, map[string]any{
			"message_id": s.messageID,
			"content":    "",
		})
		s.channel.broadcastToSession(s.chatID, createMsg)
		s.sent = true
	}

	s.channel.broadcastToSession(s.chatID, msg)
	s.lastAt = now
	return nil
}

func (s *picoStreamer) Finalize(ctx context.Context, content string) error {
	s.mu.Lock()
	defer s.mu.Unlock()

	msg := newMessage(TypeMessageUpdate, map[string]any{
		"message_id": s.messageID,
		"content":    content,
	})
	return s.channel.broadcastToSession(s.chatID, msg)
}

func (s *picoStreamer) Cancel(ctx context.Context) {
	// Nothing to clean up for WebSocket streams
}
