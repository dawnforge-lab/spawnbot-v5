// registerPending adds a PendingContinuation to the tracker and returns its ID.
func (al *AgentLoop) registerPending(pc PendingContinuation) string {
	id := fmt.Sprintf("%x", time.Now().UnixNano())
	pc.ID = id
	al.pendingConts.Store(id, &pc)
	return id
}

// unregisterPending removes a continuation from the tracker.
func (al *AgentLoop) unregisterPending(id string) {
	al.pendingConts.Delete(id)
}

// GetPendingContinuations returns all in-flight continuations for the given
// agent ID. Returns nil if none are registered.
func (al *AgentLoop) GetPendingContinuations(agentID string) []PendingContinuation {
	var result []PendingContinuation
	al.pendingConts.Range(func(_, v any) bool {
		pc := v.(*PendingContinuation)
		if agentID == "" || pc.AgentID == agentID {
			result = append(result, *pc)
		}
		return true
	})
	return result
}
```

package gateway

import (
	"encoding/json"
	"net/http"
	"time"

	"github.com/dawnforge-lab/spawnbot-v5/pkg/agent"
)

type pendingContinuationResponse struct {
	ID         string     `json:"id"`
	AgentID    string     `json:"agent_id"`
	SessionKey string     `json:"session_key"`
	Kind       string     `json:"kind"`
	Intent     string     `json:"intent"`
	CreatedAt  time.Time  `json:"created_at"`
	FiresAt    *time.Time `json:"fires_at,omitempty"`
	EventName  string     `json:"event_name,omitempty"`
}

// makeContinuationsHandler returns an HTTP handler that serves
// GET /agents/{id}/continuations for the given AgentLoop.
func makeContinuationsHandler(al *agent.AgentLoop) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		if r.Method != http.MethodGet {
			http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
			return
		}
		agentID := r.PathValue("id")
		pending := al.GetPendingContinuations(agentID)
		resp := make([]pendingContinuationResponse, 0, len(pending))
		for _, pc := range pending {
			resp = append(resp, pendingContinuationResponse{
				ID:         pc.ID,
				AgentID:    pc.AgentID,
				SessionKey: pc.SessionKey,
				Kind:       string(pc.Kind),
				Intent:     pc.Intent,
				CreatedAt:  pc.CreatedAt,
				FiresAt:    pc.FiresAt,
				EventName:  pc.EventName,
			})
		}
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(resp)
	}
}
```

package api

import (
	"encoding/json"
	"fmt"
	"net"
	"net/http"
	"strconv"
	"time"
)

// registerContinuationRoutes binds continuation proxy endpoints.
func (h *Handler) registerContinuationRoutes(mux *http.ServeMux) {
	mux.HandleFunc("GET /api/agents/{id}/continuations", h.handleGetContinuations)
}

// handleGetContinuations proxies GET /api/agents/{id}/continuations to the
// gateway at port 18790.
func (h *Handler) handleGetContinuations(w http.ResponseWriter, r *http.Request) {
	agentID := r.PathValue("id")
	gatewayURL := fmt.Sprintf("http://%s/agents/%s/continuations",
		net.JoinHostPort("127.0.0.1", strconv.Itoa(18790)),
		agentID,
	)

	client := &http.Client{Timeout: 3 * time.Second}
	resp, err := client.Get(gatewayURL)
	if err != nil {
		// Gateway unreachable — return empty list rather than error so UI stays calm.
		w.Header().Set("Content-Type", "application/json")
		w.Write([]byte("[]"))
		return
	}
	defer resp.Body.Close()

	var payload json.RawMessage
	if err := json.NewDecoder(resp.Body).Decode(&payload); err != nil {
		w.Header().Set("Content-Type", "application/json")
		w.Write([]byte("[]"))
		return
	}
	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(payload)
}
```

import { useEffect, useState } from "react"

import { getPendingContinuations, type PendingContinuation } from "@/api/continuations"

export function usePendingContinuations(agentID: string, pollIntervalMs = 5000) {
  const [pending, setPending] = useState<PendingContinuation[]>([])

  useEffect(() => {
    if (!agentID) return
    let cancelled = false

    const poll = async () => {
      try {
        const result = await getPendingContinuations(agentID)
        if (!cancelled) setPending(result)
      } catch {
        // gateway may not be running; keep previous state
      }
    }

    poll()
    const id = setInterval(poll, pollIntervalMs)
    return () => {
      cancelled = true
      clearInterval(id)
    }
  }, [agentID, pollIntervalMs])

  return pending
}
```

package agent

import (
	"context"
	"os"
	"sync/atomic"
	"testing"
	"time"

	"github.com/dawnforge-lab/spawnbot-v5/pkg/bus"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/config"
	"github.com/dawnforge-lab/spawnbot-v5/pkg/providers"
)

// scriptedProvider returns responses from a pre-loaded sequence.
type scriptedProvider struct {
	responses []providers.LLMResponse
	idx       atomic.Int32
}

func (s *scriptedProvider) Chat(
	_ context.Context,
	_ []providers.Message,
	_ []providers.ToolDefinition,
	_ string,
	_ map[string]any,
) (*providers.LLMResponse, error) {
	i := int(s.idx.Add(1)) - 1
	if i >= len(s.responses) {
		return &providers.LLMResponse{Content: "done"}, nil
	}
	r := s.responses[i]
	return &r, nil
}

func (s *scriptedProvider) GetDefaultModel() string { return "scripted-model" }

func endTurnCall(continuation, intent string) providers.ToolCall {
	return providers.ToolCall{
		Name:      "end_turn",
		Arguments: map[string]any{"continuation": continuation, "intent": intent},
	}
}

func newTestLoopWithDepth(t *testing.T, maxDepth int) (*AgentLoop, func()) {
	t.Helper()
	tmpDir, err := os.MkdirTemp("", "cont-test-*")
	if err != nil {
		t.Fatalf("MkdirTemp: %v", err)
	}
	cfg := &config.Config{
		Agents: config.AgentsConfig{
			Defaults: config.AgentDefaults{
				Workspace:            tmpDir,
				ModelName:            "scripted-model",
				MaxTokens:            4096,
				MaxToolIterations:    20,
				MaxAutoContinueDepth: maxDepth,
			},
		},
	}
	return NewAgentLoop(cfg, bus.NewMessageBus(), &scriptedProvider{}), func() {
		os.RemoveAll(tmpDir)
	}
}

// TestContinuation_ContinueNow verifies that end_turn(continue_now) triggers
// a second turn with the self-continue steering message, then end_turn(done)
// cleanly terminates the chain.
func TestContinuation_ContinueNow(t *testing.T) {
	provider := &scriptedProvider{
		responses: []providers.LLMResponse{
			{
				Content:   "I will continue.",
				ToolCalls: []providers.ToolCall{endTurnCall("continue_now", "testing continuation")},
			},
			{
				Content:   "Done now.",
				ToolCalls: []providers.ToolCall{endTurnCall("done", "finished")},
			},
		},
	}

	tmpDir := t.TempDir()
	cfg := &config.Config{
		Agents: config.AgentsConfig{
			Defaults: config.AgentDefaults{
				Workspace:            tmpDir,
				ModelName:            "scripted-model",
				MaxTokens:            4096,
				MaxToolIterations:    20,
				MaxAutoContinueDepth: 5,
			},
		},
	}
	al := NewAgentLoop(cfg, bus.NewMessageBus(), provider)

	_, err := al.processMessage(context.Background(), bus.InboundMessage{
		Channel:  "test",
		SenderID: "test:user",
		ChatID:   "test-chat-1",
		Content:  "start",
	})
	if err != nil {
		t.Fatalf("processMessage() error = %v", err)
	}

	// Give the async continuation goroutine time to run
	time.Sleep(200 * time.Millisecond)

	// Provider should have been called twice: turn 1 + continuation turn 2
	calls := int(provider.idx.Load())
	if calls < 2 {
		t.Errorf("provider called %d times, want >= 2 (turn + continuation)", calls)
	}
}

// TestContinuation_DepthCapEnforced verifies that the agent stops after
// MaxAutoContinueDepth consecutive self-continuations.
func TestContinuation_DepthCapEnforced(t *testing.T) {
	// Always respond with continue_now
	provider := &scriptedProvider{
		responses: func() []providers.LLMResponse {
			out := make([]providers.LLMResponse, 20)
			for i := range out {
				out[i] = providers.LLMResponse{
					Content:   "continuing",
					ToolCalls: []providers.ToolCall{endTurnCall("continue_now", "keep going")},
				}
			}
			return out
		}(),
	}

	tmpDir := t.TempDir()
	const cap = 3
	cfg := &config.Config{
		Agents: config.AgentsConfig{
			Defaults: config.AgentDefaults{
				Workspace:            tmpDir,
				ModelName:            "scripted-model",
				MaxTokens:            4096,
				MaxToolIterations:    20,
				MaxAutoContinueDepth: cap,
			},
		},
	}
	al := NewAgentLoop(cfg, bus.NewMessageBus(), provider)

	_, err := al.processMessage(context.Background(), bus.InboundMessage{
		Channel:  "test",
		SenderID: "test:user",
		ChatID:   "test-chat-2",
		Content:  "start",
	})
	if err != nil {
		t.Fatalf("processMessage() error = %v", err)
	}

	// Wait for the chain to exhaust
	time.Sleep(500 * time.Millisecond)

	calls := int(provider.idx.Load())
	// Should be capped at cap+1 (initial turn + cap continuations)
	if calls > cap+2 {
		t.Errorf("provider called %d times; depth cap of %d not enforced", calls, cap)
	}
}

// TestContinuation_ShutdownDrainsPending verifies that Stop() logs a warning
// for any in-flight schedule continuation and does not hang.
func TestContinuation_ShutdownDrainsPending(t *testing.T) {
	// Respond with a 1-hour schedule continuation
	provider := &scriptedProvider{
		responses: []providers.LLMResponse{
			{
				Content: "will check back",
				ToolCalls: []providers.ToolCall{{
					Name: "end_turn",
					Arguments: map[string]any{
						"continuation": "schedule",
						"intent":       "check back in an hour",
						"at":           time.Now().Add(time.Hour).Format(time.RFC3339),
					},
				}},
			},
		},
	}

	tmpDir := t.TempDir()
	cfg := &config.Config{
		Agents: config.AgentsConfig{
			Defaults: config.AgentDefaults{
				Workspace:            tmpDir,
				ModelName:            "scripted-model",
				MaxTokens:            4096,
				MaxToolIterations:    20,
				MaxAutoContinueDepth: 5,
			},
		},
	}
	al := NewAgentLoop(cfg, bus.NewMessageBus(), provider)

	_, err := al.processMessage(context.Background(), bus.InboundMessage{
		Channel:  "test",
		SenderID: "test:user",
		ChatID:   "test-chat-3",
		Content:  "remind me in an hour",
	})
	if err != nil {
		t.Fatalf("processMessage() error = %v", err)
	}

	// Give the goroutine time to register the pending continuation
	time.Sleep(100 * time.Millisecond)

	// Stop should return promptly (not hang on the 1-hour timer)
	done := make(chan struct{})
	go func() {
		al.Stop()
		close(done)
	}()

	select {
	case <-done:
		// OK
	case <-time.After(2 * time.Second):
		t.Fatal("Stop() hung; scheduleSelfContinuation did not respect done channel")
	}
}
